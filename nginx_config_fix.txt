# HTTP сервер - перенаправляет все запросы на HTTPS
server {
    listen 80;
    listen [::]:80;
    server_name qrshot.kz www.qrshot.kz;

    # Для Let's Encrypt (обновление сертификата)
    location /.well-known/acme-challenge/ {
        root /var/www/html;
    }

    # Перенаправление всех HTTP запросов на HTTPS
    return 301 https://$server_name$request_uri;
}

# HTTPS сервер
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name qrshot.kz www.qrshot.kz;

    # Корневая директория для статических файлов фронтенда
    root /var/www/qrshot/frontend;
    index index.html;

    # SSL сертификаты
    ssl_certificate /etc/letsencrypt/live/qrshot.kz/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/qrshot.kz/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    # API запросы проксируем на Node.js сервер
    location /api/ {
        proxy_pass http://localhost:5000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        
        # Увеличиваем таймауты для загрузки файлов
        proxy_read_timeout 300s;
        proxy_connect_timeout 300s;
        proxy_send_timeout 300s;
        
        # Увеличиваем размер загружаемых файлов
        client_max_body_size 100M;
    }

    # Загруженные файлы (uploads) проксируем на Node.js
    # ВАЖНО: Этот блок ДОЛЖЕН быть ПЕРЕД блоком статических файлов
    location ^~ /uploads/ {
        proxy_pass http://localhost:5000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # ВАЖНО: Сохраняем оригинальные заголовки от Express (включая Content-Type)
        proxy_pass_header Content-Type;
        proxy_ignore_headers Cache-Control Expires;
        
        # Отключаем кэширование для uploads
        proxy_hide_header Cache-Control;
        proxy_hide_header Expires;
        add_header Cache-Control "no-store, no-cache, must-revalidate, private" always;
        add_header Pragma "no-cache" always;
        add_header Expires "0" always;
        
        # Отключаем буферизацию для больших файлов
        proxy_buffering off;
        proxy_request_buffering off;
        
        # Увеличиваем таймауты для больших файлов
        proxy_read_timeout 300s;
        proxy_connect_timeout 300s;
        proxy_send_timeout 300s;
    }

    # Статические файлы (CSS, JS, изображения)
    location ~* \.(css|js|jpg|jpeg|png|gif|ico|svg|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
        try_files $uri =404;
    }

    # robots.txt (доступен для поисковых роботов)
    location = /robots.txt {
        access_log off;
    }

    # HTML файлы фронтенда
    location / {
        try_files $uri $uri/ /index.html;
        
        # Отключаем кэширование для HTML
        add_header Cache-Control "no-store, no-cache, must-revalidate";
    }
}
